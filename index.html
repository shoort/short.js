short.js

write more with less

Ever wanted to write simple stuff in JS, without all the typing and polyfills?

short.js provides elegant solutions to your problems!
Your code will be simple, with less repetition.

short.js is a work in progress! Check out our GitHub repo!

I/O Made Easy

// Less typing, more printing!
io.log('Hello, world!');
io.warn('Something might be up! Be wary!');
io.error('I was right! Something is wrong!');

// Some stuff is only useful to developers.
// Hide it in production by removing one line!
io.debug('This will not be logged!');
io.debug(true);
io.debug('This will be logged!');

// Alerts an­d prompts and confirms - oh my!
io.alert('Hello, there! Sorry to interrupt!');
var love = io.confirm('Do you love me?');
var card = io.prompt('Pick a card!');

// Web requests are a piece of cake!
io.get('/foo.json', function (res) {
  io.log('HTTP Status: ' + res.status);
  io.log('Response Text: ' + res.text);
  io.log('Response JSON: ' + res.json);
});

DOM is No Longer Dumb

// Quickly find elements with a shorthand for querySelector!
var logger = dom.qs('#logger');

// Bind events to functions easily!
logger.click(io.log, 'You clicked the logger!');

// Remember when it took work to hide and show things?
// Now you can toggle classes easily!
var switch = dom.qs('#switch');
var box = dom.qs('#box');
switch.click(box.classes.toggle, 'show');

Iteration for All

// Now, you can iterate through element lists!
var links = dom.qsa('a');

links.each(function (link) {
  link.click(void);
});

// Iterate through objects, too!
var toppings = {
  tomato: true,
  anchovies: false,
  pepperoni: true
};

io.log('I want a large pizza please...');

toppings.each(function (key, value) {
  io.log((value ? 'With extra ' : 'Hold the ') + key);
});

// Looping a certain number of times is suddenly a lot easier! Wow!
times(7, function () {
  io.log("What's new, pussycat?");
});
io.log("It's not unusual.");

// Cool! Ranges are a thing now!
[3, 10].range();
// -> [3, 4, 5, 6, 7, 8, 9, 10]
